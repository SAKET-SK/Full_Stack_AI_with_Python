CREATE(u:Company {name:"OMFYS"})

Establish relationship

MERGE(u:User {name:"Saket"}) - [:EMPLOYEE] -> (c:Company {name:"OMFYS"})

MATCH(u:User {name:"Saket"})

MERGE(u) - [:EMPLOYEE]->(c)
